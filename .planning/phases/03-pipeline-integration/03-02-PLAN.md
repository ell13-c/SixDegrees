# SC-3: center user at (0.0, 0.0)
center = [r for r in rows.data if r['other_user_id'] == USER_ID]
print(f'SC-3 center user row: {center}')
assert len(center) == 1, 'Center user not in coordinates'
assert center[0]['x'] == 0.0 and center[0]['y'] == 0.0, f'Center not at origin: {center[0]}'
print('SC-3 PASSED: center user at (0.0, 0.0)')

# SC-2: run again, verify old rows marked is_current=False and new rows inserted
run_pipeline_for_user(USER_ID)
all_rows = sb.table('map_coordinates').select('is_current').eq('center_user_id', USER_ID).execute()
current = [r for r in all_rows.data if r['is_current']]
old = [r for r in all_rows.data if not r['is_current']]
print(f'SC-2 after second run — current: {len(current)}, old retained: {len(old)}')
assert len(old) >= 1, 'Old rows were deleted (STORE-02 violation)'
assert len(current) >= 10, 'Too few current rows after second run'
print('SC-2 PASSED: old rows retained, new rows current')
print('ALL PHASE 3 ROADMAP SUCCESS CRITERIA PASSED')
